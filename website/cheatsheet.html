Basic Shapes

2D Shapes

-- Rectangle/Square
cad.rect({size = {width, height}, center = false})
cad.square({size = width, center = false})

-- Circle
cad.circle({r = radius})
cad.circle({d = diameter})

-- Polygon
cad.polygon({points = {{x1,y1}, {x2,y2}, ...}})

3D Shapes

-- Cube/Cuboid
cad.cube({size = {x,y,z}, center = false})

-- Sphere
cad.sphere({r = radius})
cad.sphere({d = diameter})

-- Cylinder/Cone
cad.cylinder({h = height, r = radius, center = false})
cad.cylinder({h = height, r1 = bottom_r, r2 = top_r})

-- Polyhedron
cad.polyhedron(x, y, z, points, faces)

Special Shapes

2D Special Shapes

-- Rounded Rectangle
cad.rectround({size = {w,h}, radius = r})

-- Chamfered Rectangle
cad.rectchamfer({size = {w,h}, chamfer = c})

-- Long Hole (Slot)
cad.longhole({points = {{x1,y1}, {x2,y2}}, r = radius})
cad.longholerelative({point = {x,y}, delta = {dx,dy}, r = radius})

3D Special Shapes

-- Pyramid
cad.pyramid(x, y, z, length, height)

-- Text
cad.text(x, y, z, "text", height, font, style)
cad.text3d(x, y, z, "text", height, depth, font)

Transformations

-- Translate
object:translate(x, y, z)

-- Rotate
object:rotate(cx, cy, cz, x_angle, y_angle, z_angle)
object:rotate2d(cx, cy, angle)

-- Scale
object:scale(x, y, z)

-- Mirror
object:mirror(x, y, z)

-- Resize
object:resize(x, y, z, auto_x, auto_y, auto_z)

Boolean Operations

-- Union
a + b                   -- Operator syntax
object:add(obj1, obj2)  -- Method syntax
object:union()          -- Combined elements

-- Difference
a - b
object:sub(obj1, obj2)

-- Intersection
object:intersect(obj1, obj2)

Extrusions

-- Linear extrude
object:linearextrude(depth, twist, slices, scale)

-- Rotate extrude
object:rotateextrude()

-- Hull
object:hull()

-- Offset (2D)
object:offset(distance, chamfer)
object:offsetradius(radius, chamfer)

Object Operations

-- Create new object
local obj = cad()

-- Copy/Clone
local copy = obj:copy()
local clone = obj:clone()

-- Reset
obj:init()
obj:reset()

-- Set color
obj:setcolor(r, g, b)
obj:setcolor("red")
obj:setcolor({r, g, b})

-- Import/Export
obj:import("model.stl")
obj:export("output.stl", color, verbose)
cad.export("combined.stl", obj1, obj2, obj3)

Vector Operations

-- Create vector
local v = vector(x, y, z)
local v = vec(x, y, z)

-- Vector operations
v1 + v2           -- Addition
v1 - v2           -- Subtraction
5 * v or v * 5    -- Scalar multiplication
-v                -- Negation
#v or v:len()     -- Length

-- Vector methods
v:getx(), v:gety(), v:getz()  -- Get components
v:unit()                      -- Get unit vector
v:normal(direction)           -- Get normal vector
v:scalar(v2)                  -- Calculate dot product
v:cross(v2)                   -- Calculate cross product
v:rot(phi, axis)              -- Rotate vector
v:angle(v2)                   -- Angle between vectors

DXF Operations

-- Create DXF object
local drawing = dxf()

-- DXF methods
drawing:addlayer(name, colorindex, style)
drawing:text(text, x, y, height, angle, font, thickness, layer)
drawing:line(x1, y1, x2, y2, layer)
drawing:circle(x, y, radius, layer)
drawing:arc(x, y, radius, start_angle, end_angle, layer)
drawing:rect(x1, y1, w, h, layer)
drawing:polygon(args)
drawing:save("output.dxf")

Geometry Utilities

-- Process polygon points with special functions
geometry.polygon(points)

-- Special point functions
-- Format: {x, y, "funcname", {args}}
{x, y, "radius", {radius, segments}}
{x, y, "belly", {distance, segments}}
{x, y, "arc", {radius, segments}}

-- Generate arc points
geometry.arc(x, y, radius, start_angle, end_angle, segments)

-- Generate circular segment
geometry.getcirclesegment(x1, y1, x2, y2, dist, segments)
